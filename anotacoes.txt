# Esse arquivo txt é para a anotação dos codigos a serem aprendidos no 
curso da DIO

- git status  //mostra a situação atual do diretorio 
- git commit -m "Titulo" //serve para colocar titulo 
- git push origin main  //manda todos os arquivos alterados e com titulo
- git add .  //adiciona todos os arquivos que não estão no repositorio
- git clone  //serve para clonar um repositorio 
- cd /c  //serve para ir em uma pasta especifica da maquina
- cd ..  //serve para retroceder a pasta atual
- ls  //uma lista de todas as pastas 
- ls -la //uma lista de todas as pastas inclusive as ocultas
- git remote -v  //origem dos repositorios 
- git config --list  //mostra a lista de configurações do github
- git init //inicia um repositorio no github
